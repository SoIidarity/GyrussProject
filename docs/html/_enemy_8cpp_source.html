199 void Enemy::move()
200 {
201  // Store old position
202  _prevPosition = _sprite.getPosition();
203  _prevPosition.x -= _resolution.x/2;
204  _prevPosition.y -= _resolution.y/2;
205 
206  _angle += _futureAngleValue;
207  _angle = common::angleFilter(_angle);
208  auto offset = 0.f;
209  if(_distanceFromCentre==0)
210  {
211  offset = float(_resolution.x*0.3);
212  }
213  auto depthScale = ((_distanceFromCentre + offset)/(_resolution.x/2));
214  _distanceFromCentre += (_futureMoveValue) * depthScale;
215  auto xPos = _distanceFromCentre * sin(common::degreeToRad(_angle))+_centre.x;
216  auto yPos = _distanceFromCentre * cos(common::degreeToRad(_angle))+_centre.y;
217  auto scale = 1 + ((getRadius() - (_resolution.x / 2)) / (_resolution.x / 2));
218 
219  _sprite.setPosition(xPos+(_resolution.x / 2),yPos+(_resolution.y / 2));
220  _sprite.setScale(scale * _scale,scale * _scale);
221  // Dimming
222  auto dimColor = (scale*55) + 200;
223  _sprite.setColor(Color(dimColor,dimColor,dimColor));
224 
225  setOrientation();
226 }
