208 void Enemy::move()
209 {
210  // Store old position
211  _prevPosition = _sprite.getPosition();
212  _prevPosition.x -= _resolution.x/2;
213  _prevPosition.y -= _resolution.y/2;
214 
215  _angle += _futureAngleValue;
216  _angle = common::angleFilter(_angle);
217  auto offset = 0.f;
218  if(_distanceFromCentre==0)
219  {
220  offset = _resolution.x*0.3;
221  }
222  auto depthScale = ((_distanceFromCentre + offset)/(_resolution.x/2));
223  _distanceFromCentre += (_futureMoveValue) * depthScale;
224  auto xPos = _distanceFromCentre * sin(common::degreeToRad(_angle))+_centre.x;
225  auto yPos = _distanceFromCentre * cos(common::degreeToRad(_angle))+_centre.y;
226  auto scale = 1 + ((getRadius() - (_resolution.x / 2)) / (_resolution.x / 2));
227 
228  _sprite.setPosition(xPos+(_resolution.x / 2),yPos+(_resolution.y / 2));
229  _sprite.setScale(scale * _scale,scale * _scale);
230  // Dimming
231  auto dimColor = (scale*55) + 200;
232  _sprite.setColor(sf::Color(dimColor,dimColor,dimColor));
233 
234  // Orientation
235  _newPosition = _sprite.getPosition();
236  _newPosition.x -= _resolution.x/2;
237  _newPosition.y -= _resolution.y/2;
238  _pointingPosition = _newPosition - _prevPosition;
239  _angleOrientation = _futureAngleValue = atan2(_pointingPosition.x,_pointingPosition.y) - atan2(_prevPosition.x,_prevPosition.y); // ? Should be -?
240  _angleOrientation = -1*common::radToDegree(_angleOrientation) - getAngleWithOffset();
241  _sprite.setRotation(_angleOrientation);
242 }
