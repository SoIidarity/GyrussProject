198 void Enemy::move()
199 {
200  // Store old position
201  _prevPosition = _sprite.getPosition();
202  _prevPosition.x -= _resolution.x/2;
203  _prevPosition.y -= _resolution.y/2;
204 
205  _angle += _futureAngleValue;
206  _angle = common::angleFilter(_angle);
207  auto offset = 0.f;
208  if(_distanceFromCentre==0)
209  {
210  offset = float(_resolution.x*0.3);
211  }
212  auto depthScale = ((_distanceFromCentre + offset)/(_resolution.x/2));
213  _distanceFromCentre += (_futureMoveValue) * depthScale;
214  auto xPos = _distanceFromCentre * sin(common::degreeToRad(_angle))+_centre.x;
215  auto yPos = _distanceFromCentre * cos(common::degreeToRad(_angle))+_centre.y;
216  auto scale = 1 + ((getRadius() - (_resolution.x / 2)) / (_resolution.x / 2));
217 
218  _sprite.setPosition(xPos+(_resolution.x / 2),yPos+(_resolution.y / 2));
219  _sprite.setScale(scale * _scale,scale * _scale);
220  // Dimming
221  auto dimColor = (scale*55) + 200;
222  _sprite.setColor(Color(dimColor,dimColor,dimColor));
223 
224  setOrientation();
225 }

227 void Enemy::setOrientation()
228 {
229  // Orientation
230  _newPosition = _sprite.getPosition();
231  _newPosition.x -= _resolution.x/2;
232  _newPosition.y -= _resolution.y/2;
233  _pointingPosition = _newPosition - _prevPosition;
234  _angleOrientation = _futureAngleValue = atan2(_pointingPosition.x,_pointingPosition.y) - atan2(_prevPosition.x,_prevPosition.y); // ? Should be -?
235  _angleOrientation = -1*common::radToDegree(_angleOrientation) - getAngleWithOffset();
236  _sprite.setRotation(_angleOrientation);
