49 void Game::startGameLoop()
50 {
51  Clock mainClock;
52  Clock totalTimer;

56  float timeStep = 1.f / 60.f; //60 frames per second
57  auto speedModifier = 0.5f; //defines how fast the game should be at the start
58  auto increaseSpeedThreshold = 1.5f; // How often the game speeds up (in seconds)
59  auto globalSpeedIncrease = 0.02f; // Game speeds up this amount every time
60  _inputHandler.reset(); //clear all previous inputs in the inputHandler
61 
62  _mainWindow.clear(sf::Color::Black);
63 
64  //First Game State
65  if (_gameState == game::GameState::Splash)
66  showSplashScreen();
67 
68  if (_gameState == game::GameState::GameOverLose)
69  showGameOverScreen(false);
70 
71  if (_gameState == game::GameState::GameOverWin)
72  showGameOverScreen(true);
73 
74  #ifdef DEBUG_ONLY
75  // Debug mode: displays frames-per-second in title bar
76  FPS fps;
77  #endif // DEBUG_ONLY
78 
82  _score.reset();
83 
84  // Load game music
85  if(_soundController.loadMusic())
86  _soundController.playMusic();
87 
88  // Generate starField
89  auto numberOfStars = 60;
90  StarField starField(_resolution, 3, numberOfStars);
91 
92  // Generate playerShip object
93  auto shipPathRadiusPadding = 0.05f;
94  const auto shipPathRadius = (_resolution.y / 2) - (_resolution.y * shipPathRadiusPadding);
95  const auto shipScale = 0.22;
96  PlayerShip playerShip(_resolution,
97  shipPathRadius,
98  0,
99  shipScale,
100  entity::PlayerShip,
101  _textures);
102 
103  // Generate entityController object to manage all game objects movement, creation and destruction
104  EntityController entityController(_resolution,
105  playerShip,
106  _textures,
107  _score,
108  speedModifier);
109 
110  // Generate shield object for playerShip
111  Shield shield(_resolution,
112  shipPathRadius,
113  0,
114  shipScale,
115  _textures,
116  playerShip);
117 
118  // Create a hud object to show current lives and game stats
119  HUD hud(_resolution,
120  _mainWindow,
121  _fonts,
122  _score,
123  playerShip);
124 
125  // Initialise the playerShip move sound
126  _soundController.playSound(sounds::PlayerMove,0,100,1);
127 
131  while (_gameState == game::GameState::Playing)
132  {
136  // Deal with user input
137  Event event;
138  while (_mainWindow.pollEvent(event))
139  {
140  _inputHandler.pollInput(_gameState, playerShip, event);
141 
142  #ifdef DEBUG_ONLY
143  // Debug function (developer cheat keys to tweak and test game-play)
144  debugKeys(event, playerShip, entityController);
145  #endif // DEBUG_ONLY
146  }
147 
148  // Frame timing events
149  timeSinceUpdate += mainClock.getElapsedTime();
150  mainClock.restart();
151 
152  // Game gets faster the longer you stay alive
153  if(speedTimer.getElapsedTime().asSeconds() > increaseSpeedThreshold)
154  {
155  speedTimer.restart();
156  entityController.changeGlobalSpeed(globalSpeedIncrease);
157  }
158 
162  while (timeSinceUpdate.asSeconds() >= timeStep)
163  {
164  timeSinceUpdate = sf::Time::Zero;
165 
166  entityController.spawnEntities();
167  _inputHandler.setPlayerShipMove(playerShip, timeStep);
168  entityController.setMove();
169  entityController.shoot();
170  entityController.checkClipping();
171 
175 
176  // Returns true if the player has collided. (also does global entity collision check)
177  if (entityController.checkCollisions())
178  {
179  if (!playerShip.isInvulnerable())
180  {
181  playerDeathEvents(playerShip, entityController);
182  }
183  }
184 
188 
189  playEventSounds(playerShip, entityController);
190 
194  update(playerShip, entityController);
195 
199  render(starField, playerShip, entityController, shield, hud);
200 
201  #ifdef DEBUG_ONLY
202  fps.update();
203  std::ostringstream ss;
204  ss << fps.getFPS();
205  _mainWindow.setTitle(ss.str());
206  #endif // DEBUG_ONLY
207 
211  endGameCheck(playerShip);
212 
213  }
214  }
