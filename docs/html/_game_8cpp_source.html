219 void Game::pollInput(PlayerShip& playerShip,
220  std::vector<Bullet>& bulletVector,
221  sf::Event &event,
222  bool &previousButtonState)
223 {
224  // ToDo: Move into inputHandler
225 
226  //Check for events since last frame
227  while (_mainWindow.pollEvent(event))
228  {
229  if (event.type == sf::Event::Closed)
230  {
231  _gameState = Game::Exiting;
232  }
233 
234  if (sf::Keyboard::isKeyPressed(sf::Keyboard::Q)
235  && (sf::Keyboard::isKeyPressed(sf::Keyboard::LControl)
236  || sf::Keyboard::isKeyPressed(sf::Keyboard::RControl)))
237  {
238  _gameState = Game::Exiting;
239  }
240  if (event.type == sf::Event::EventType::KeyPressed)
241  if (event.key.code == sf::Keyboard::Space)
242  if (previousButtonState == 0)
243  {
244  playerShip.shoot();
245  Bullet bullet(_textures,_resolution,
246  playerShip.getDistanceFromCentre(),
247  playerShip.getAngle(), 1 , textures::BulletPlayer);
248  bulletVector.push_back(bullet);
249  // ToDo : Make bullet creation an event
250  previousButtonState = 1;
251  }
252  if (event.type == sf::Event::EventType::KeyReleased)
253  if (event.key.code == sf::Keyboard::Space)
254  {
255  previousButtonState = 0;
256  }
257 
258  // During the current polling period, key-presses are detected
259  // if pressed added to the map, and removed if the key is released
260  if (event.type == sf::Event::EventType::KeyPressed)
261  {
262  if (_keysPressed.count(event.key.code) == 0)
263  {
264  _keysPressed[event.key.code] = true;
265  }
266  }
267 
268  if (event.type == sf::Event::EventType::KeyReleased)
269  {
270  if (_keysPressed.count(event.key.code) > 0)
271  {
272  _keysPressed.erase(event.key.code);
273  }
274  }
275  } // End of input polling
