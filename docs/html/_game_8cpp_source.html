59  auto increaseSpeedThreshold = 1.5f; // How often the game speeds up (in seconds)
60  auto globalSpeedIncrease = 0.025f; // Game speeds up this amount every time that it increases
61  _inputHandler.reset(); // Clear all previous inputs in the inputHandler

88 
89  // Generate starField
90  auto numberOfStars = 60;
91  StarField starField(_resolution, 3, numberOfStars);

123  _score,
124  playerShip);

126  // Initialise the playerShip move sound

148

150  timeSinceUpdate += mainClock.getElapsedTime();
151  mainClock.restart();
152 
153  // Game gets faster the longer you stay alive
154  if(speedTimer.getElapsedTime().asSeconds() > increaseSpeedThreshold)
