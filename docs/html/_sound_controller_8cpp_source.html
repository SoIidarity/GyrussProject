57 void SoundController::getSounds()
58 {
59  // Constructs the vector of sound objects by iterating through the enum
60  // This relies upon the dummy enum entry soundID_end, and the first entry to
61  // the enum list being assigned to 0, with the remaining entries in the enum list
62  // in default (non-specified) numerical order
63  sounds::ID id;
64  int end = (sounds::soundID_end-1);
65  for(auto i = 0; i < end ; i++)
66  {
67  id = (sounds::ID)i;
68  sf::Sound sound;
69  _soundList.push_back(sound);
70  _soundList.at(i).setBuffer(_sounds.get(id));
71  }
72 }
