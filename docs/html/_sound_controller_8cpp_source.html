61 void SoundController::getSounds()
62 {
63  // Constructs the vector of sound objects by iterating through the enum
64  // This relies upon the dummy enum entry soundID_end, and the first entry to
65  // the enum list being assigned to 0, with the remaining entries in the enum list
66  // in default (non-specified) numerical order
67  sounds::ID id;
68  int end = (sounds::soundID_end-1);
69  for(auto i = 0; i < end ; i++)
70  {
71  id = (sounds::ID)i;
72  sf::Sound sound;
73  _soundList.push_back(sound);
74  _soundList.at(i).setBuffer(_sounds.get(id));
75  }
76 }
