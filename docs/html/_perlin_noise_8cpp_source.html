37 double PerlinNoise::noise(double x, double y , double z) {
38  // Find the unit cube that contains the point
39  int X = (int) floor(x) & 255;
40  int Y = (int) floor(y) & 255;
41  int Z = (int) floor(z) & 255;
42 
43  // Find relative x, y,z of point in cube
44  x -= floor(x);
45  y -= floor(y);
46  z -= floor(z);
47 
48  // Compute fade curves for each of x, y, z
49  double u = smootherStep(x);
50  double v = smootherStep(y);
51  double w = smootherStep(z);
52 
53  // Hash coordinates of the 8 cube corners
54  int A = permutationVector[X] + Y;
55  int AA = permutationVector[A] + Z;
56  int AB = permutationVector[A + 1] + Z;
57  int B = permutationVector[X + 1] + Y;
58  int BA = permutationVector[B] + Z;
59  int BB = permutationVector[B + 1] + Z;
60 
61  // Add blended results from 8 corners of cube
62  double res = lerp(w,
63  lerp(v,
64  lerp(u, grad(permutationVector[AA], x, y, z),
65  grad(permutationVector[BA], x-1, y, z)),
66  lerp(u, grad(permutationVector[AB], x, y-1, z),
67  grad(permutationVector[BB], x-1, y-1, z))),
68  lerp(v,
69  lerp(u, grad(permutationVector[AA+1], x, y, z-1),
70  grad(permutationVector[BA+1], x-1, y, z-1)),
71  lerp(u, grad(permutationVector[AB+1], x, y-1, z-1),
72  grad(permutationVector[BB+1], x-1, y-1, z-1))));
73  return (res + 1.0)/2.0;
74 }

89 double PerlinNoise::grad(int hash, double x, double y, double z)
90 {
91  int h = hash & 15;
92  // Convert lower 4 bits of hash into 12 gradient directions
93  double u = h < 8 ? x : y;
94  double v = h < 4 ? y : h == 12 || h == 14 ? x : z;
95  return ((h & 1) == 0 ? u : -u) + ((h & 2) == 0 ? v : -v);
96 }
